class Solution {
    public int[] createTargetArray(int[] nums, int[] index) {
        List<Integer> list = new ArrayList<Integer>();
		for (int i = 0; i < nums.length; i++) {
			list.add(index[i], nums[i]);
		}
		int temp[] = new int[list.size()];
        for(int i=0;i<list.size();i++)
        {
            temp[i] = list.get(i);
        }
        return temp;
    }
}
